5️⃣ COMPLETE BUTTON LOGIC (SERVER SIDE)
Pseudo Logic
IF user.role == 'Associate' AND status == 'Draft'
    SET prepared_by_user_id = current_user
    SET prepared_at = GETUTCDATE()
    SET status = 'Prepared'

ELSE IF user.role == 'Manager' AND status == 'Prepared'
    SET reviewed_by_user_id = current_user
    SET reviewed_at = GETUTCDATE()
    SET status = 'Reviewed'

ELSE IF user.role == 'Partner' AND status == 'Reviewed'
    SET approved_by_user_id = current_user
    SET approved_at = GETUTCDATE()
    SET status = 'Approved'
    LOCK page (read-only)

ELSE
    THROW authorization error
