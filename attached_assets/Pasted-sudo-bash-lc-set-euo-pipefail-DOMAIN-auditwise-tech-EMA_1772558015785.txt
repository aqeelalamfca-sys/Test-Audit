sudo bash -lc '
set -euo pipefail

DOMAIN="auditwise.tech"
EMAIL="aqeelalam2010@gmail.com"
APP_DIR="/opt/auditwise"
REPO="https://github.com/aqeelalamfca-sys/Test-Audit"

# 1) System deps
apt-get update -y
apt-get install -y ca-certificates curl git nginx ufw

# 2) Docker (Engine + Compose v2)
if ! command -v docker >/dev/null 2>&1; then
  curl -fsSL https://get.docker.com | sh
fi
systemctl enable --now docker

# 3) Firewall (safe defaults)
ufw allow OpenSSH >/dev/null 2>&1 || true
ufw allow 80/tcp >/dev/null 2>&1 || true
ufw allow 443/tcp >/dev/null 2>&1 || true
ufw --force enable >/dev/null 2>&1 || true

# 4) Fetch code
rm -rf "$APP_DIR"
mkdir -p "$APP_DIR"
git clone "$REPO" "$APP_DIR"
cd "$APP_DIR"

# 5) Create production .env (random strong secrets)
DB_PASSWORD="$(openssl rand -hex 24)"
SESSION_SECRET="$(openssl rand -hex 32)"
WEBHOOK_SECRET="$(openssl rand -hex 32)"

cat > .env <<EOF
# Generated by one-shot deploy
NODE_ENV=production
PORT=5000
DB_PASSWORD=$DB_PASSWORD
SESSION_SECRET=$SESSION_SECRET
WEBHOOK_SECRET=$WEBHOOK_SECRET
DATABASE_URL=postgresql://auditwise:${DB_PASSWORD}@postgres:5432/auditwise?schema=public
EOF

# 6) Create docker-compose.yml (repo does not include one)
cat > docker-compose.yml <<EOF
services:
  postgres:
    image: postgres:16
    container_name: auditwise-postgres
    restart: unless-stopped
    environment:
      POSTGRES_DB: auditwise
      POSTGRES_USER: auditwise
      POSTGRES_PASSWORD: \${DB_PASSWORD}
    volumes:
      - auditwise_pgdata:/var/lib/postgresql/data

  app:
    build: .
    container_name: auditwise-app
    restart: unless-stopped
    depends_on:
      - postgres
    env_file:
      - .env
    ports:
      - "127.0.0.1:5000:5000"
    volumes:
      - ./uploads:/app/uploads

volumes:
  auditwise_pgdata:
EOF

# 7) Bring up app (builds dist and starts node dist/index.cjs)
docker compose up -d --build

# 8) Nginx reverse proxy to localhost:5000
rm -f /etc/nginx/sites-enabled/default || true
cat > /etc/nginx/sites-available/auditwise <<EOF
server {
  listen 80;
  server_name ${DOMAIN} www.${DOMAIN};

  client_max_body_size 50M;

  location / {
    proxy_pass http://127.0.0.1:5000;
    proxy_http_version 1.1;
    proxy_set_header Host \$host;
    proxy_set_header X-Real-IP \$remote_addr;
    proxy_set_header X-Forwarded-For \$proxy_add_x_forwarded_for;
    proxy_set_header X-Forwarded-Proto \$scheme;
  }
}
EOF

ln -sf /etc/nginx/sites-available/auditwise /etc/nginx/sites-enabled/auditwise
nginx -t
systemctl reload nginx

# 9) SSL (Let’s Encrypt) + auto-redirect to HTTPS
apt-get install -y certbot python3-certbot-nginx
certbot --nginx -d ${DOMAIN} -d www.${DOMAIN} --non-interactive --agree-tos -m ${EMAIL} --redirect
systemctl reload nginx

echo ""
echo "✅ LIVE: https://${DOMAIN}/"
echo "🔐 DB_PASSWORD: ${DB_PASSWORD}"
echo "🔐 SESSION_SECRET: ${SESSION_SECRET}"
echo "🔐 WEBHOOK_SECRET: ${WEBHOOK_SECRET}"
echo ""
echo "Useful checks:"
echo "  docker compose ps"
echo "  docker compose logs -f app"
'
