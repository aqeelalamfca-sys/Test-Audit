## 2) Full RBAC table (ready for migration)

### Objects

* PLATFORM: Firms, Subscriptions, Billing, Global AI Key
* FIRM: Users, Roles, Settings, Firm Wide Controls
* ENGAGEMENT: Engagements, Phases, Workpapers, Evidence, Exports, EQCR

### Actions

`VIEW, CREATE, EDIT, DELETE, ASSIGN, COMPLETE, APPROVE, EXPORT, ADMIN`

```sql
-- RBAC seed model: Permissions = (object_code, action_code)
-- Role permissions mapped via role_permissions

-- ===== Permission catalogue =====
-- Platform
-- PLATFORM_FIRM: VIEW/CREATE/EDIT/DELETE/ADMIN
-- PLATFORM_BILLING: VIEW/EDIT/ADMIN
-- PLATFORM_AI: VIEW/EDIT/ADMIN

-- Firm
-- FIRM_USER: VIEW/CREATE/EDIT/DELETE/ASSIGN/ADMIN
-- FIRM_ROLE: VIEW/CREATE/EDIT/DELETE/ADMIN
-- FIRM_SETTINGS: VIEW/EDIT/ADMIN
-- FWC: VIEW/CREATE/EDIT/DELETE/COMPLETE/APPROVE/EXPORT

-- PHASE: VIEW/EDIT/COMPLETE/APPROVE
-- WORKPAPER: VIEW/CREATE/EDIT/DELETE/COMPLETE/APPROVE/EXPORT
-- EVIDENCE: VIEW/CREATE/DELETE/EXPORT
-- EQCR: VIEW/ASSIGN/COMPLETE
-- EXPORT: CREATE/VIEW

-- ===== Role matrix (human-readable) =====
-- SUPER_ADMIN (platform only; NO firm data)
-- PLATFORM_FIRM: VIEW,CREATE,EDIT,DELETE,ADMIN
-- PLATFORM_BILLING: VIEW,EDIT,ADMIN
-- PLATFORM_AI: VIEW,EDIT,ADMIN

-- FIRM_ADMIN (firm-wide; can administer firm; can see firm data)
-- FIRM_USER: VIEW,CREATE,EDIT,DELETE,ASSIGN,ADMIN
-- FIRM_ROLE: VIEW,CREATE,EDIT,DELETE,ADMIN
-- FIRM_SETTINGS: VIEW,EDIT,ADMIN
-- FWC: VIEW,CREATE,EDIT,DELETE,COMPLETE,APPROVE,EXPORT
-- ENGAGEMENT: VIEW,CREATE,EDIT,DELETE,ASSIGN,COMPLETE,APPROVE,EXPORT
-- PHASE: VIEW,EDIT,COMPLETE,APPROVE
-- WORKPAPER: VIEW,CREATE,EDIT,DELETE,COMPLETE,APPROVE,EXPORT
-- EVIDENCE: VIEW,CREATE,DELETE,EXPORT
-- EQCR: VIEW,ASSIGN,COMPLETE
-- EXPORT: CREATE,VIEW

-- PARTNER (approve + lock)
-- FWC: VIEW,EDIT,COMPLETE,APPROVE,EXPORT
-- ENGAGEMENT: VIEW,EDIT,ASSIGN,COMPLETE,APPROVE,EXPORT
-- PHASE: VIEW,EDIT,COMPLETE,APPROVE
-- WORKPAPER: VIEW,EDIT,COMPLETE,APPROVE,EXPORT
-- EVIDENCE: VIEW,EXPORT
-- EQCR: VIEW,ASSIGN (optional),COMPLETE (if eqcr_user)
-- EXPORT: CREATE,VIEW

-- MANAGER (reviewer)
-- FWC: VIEW,EDIT,COMPLETE (review stage),EXPORT
-- ENGAGEMENT: VIEW,EDIT,ASSIGN,COMPLETE (review stage),EXPORT

-- WORKPAPER: VIEW,CREATE,EDIT,COMPLETE (review stage),EXPORT

-- PHASE: VIEW,EDIT,COMPLETE (prepared stage)
-- WORKPAPER: VIEW,CREATE,EDIT,DELETE (own drafts),COMPLETE (prepared stage)
-- EVIDENCE: VIEW,CREATE,DELETE (own uploads)
-- EXPORT: VIEW (optional)
```

## 3) Status transition flowchart (Mermaid)

```mermaid
flowchart LR
  D[Draft] -->|Complete by Associate| P[Prepared]
  P -->|Complete by Manager| R[Reviewed]
  R -->|Complete by Partner| A[Approved]
  A --> L[Locked]

  D -->|Edit allowed| D
  P -->|Edit allowed (Associate/Manager)| P
  R -->|Edit allowed (Manager/Partner)| R

  L -->|No edits| L

  A -->|Partner override with reason + audit log| R
  R -->|Manager sends back to preparer with note| P
  P -->|Preparer reworks and re-completes| P
```

**Rule:** “Complete” button logs sign-off based on role; system derives next state.

## 5) Database DDL script (PostgreSQL)

> This is a strong “baseline DDL” (multi-tenant by firm_id). You can shift to **schema-per-firm** later if desired.

```sql
-- Enable UUID
create extension if not exists "uuid-ossp";

-- ========= PLATFORM =========

create table platform_firms (
  id uuid primary key default uuid_generate_v4(),
  firm_code text not null unique,
  firm_name text not null,
  status text not null default 'ACTIVE' check (status in ('ACTIVE','SUSPENDED','DORMANT','DELETED')),
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create table platform_subscriptions (
  id uuid primary key default uuid_generate_v4(),
  firm_id uuid not null references platform_firms(id),
  plan_code text not null,
  status text not null default 'TRIAL' check (status in ('TRIAL','ACTIVE','PAST_DUE','SUSPENDED','CANCELLED')),
  trial_start date,
  trial_end date,
  active_start date,
  active_end date,
  users_limit int not null default 10,
  feature_ai_enabled boolean not null default true,
  created_at timestamptz not null default now()
);

create table platform_users (
  id uuid primary key default uuid_generate_v4(),
  firm_id uuid references platform_firms(id), -- NULL for SUPER_ADMIN if you prefer platform-only account
  email text not null unique,
  password_hash text not null,
  status text not null default 'ACTIVE' check (status in ('ACTIVE','INVITED','LOCKED','DISABLED')),
  last_login_at timestamptz,
  created_at timestamptz not null default now()
);

create table platform_roles (
  id uuid primary key default uuid_generate_v4(),
  firm_id uuid references platform_firms(id), -- NULL for platform system roles
  role_code text not null,
  role_name text not null,
  is_system boolean not null default false,
  unique (firm_id, role_code)
);

create table platform_permissions (
  id uuid primary key default uuid_generate_v4(),
  object_code text not null,
  action_code text not null,
  unique (object_code, action_code)
);

create table platform_role_permissions (
  id uuid primary key default uuid_generate_v4(),
  role_id uuid not null references platform_roles(id) on delete cascade,
  permission_id uuid not null references platform_permissions(id) on delete cascade,
  unique(role_id, permission_id)
);

create table firm_user_roles (
  id uuid primary key default uuid_generate_v4(),
  firm_id uuid not null references platform_firms(id),
  user_id uuid not null references platform_users(id) on delete cascade,
  role_id uuid not null references platform_roles(id) on delete cascade,
  unique(user_id, role_id)
);

create table platform_audit_logs (
  id uuid primary key default uuid_generate_v4(),
  firm_id uuid references platform_firms(id),
  user_id uuid references platform_users(id),
  action text not null,
  object_type text not null,
  object_id uuid,
  meta jsonb,
  ip_address inet,
  created_at timestamptz not null default now()
);

-- ========= ENGAGEMENT =========

create table audit_engagements (
  id uuid primary key default uuid_generate_v4(),
  firm_id uuid not null references platform_firms(id),
  engagement_code text not null,
  client_name text not null,
  period_start date not null,
  period_end date not null,
  status text not null default 'DRAFT' check (status in ('DRAFT','IN_PROGRESS','COMPLETED','LOCKED','ARCHIVED')),
  partner_user_id uuid references platform_users(id),
  manager_user_id uuid references platform_users(id),
  created_at timestamptz not null default now(),
  unique (firm_id, engagement_code)
);

create table audit_phases (
  id uuid primary key default uuid_generate_v4(),
  engagement_id uuid not null references audit_engagements(id) on delete cascade,
  phase_code text not null check (phase_code in ('PREPLANNING','PLANNING','RISK','EXECUTION','COMPLETION')),
  status text not null default 'DRAFT' check (status in ('DRAFT','PREPARED','REVIEWED','APPROVED','LOCKED')),
  updated_at timestamptz not null default now(),
  unique (engagement_id, phase_code)
);

create table audit_workpapers (
  id uuid primary key default uuid_generate_v4(),
  engagement_id uuid not null references audit_engagements(id) on delete cascade,
  phase_id uuid not null references audit_phases(id) on delete cascade,
  wp_code text not null,
  title text not null,
  status text not null default 'DRAFT' check (status in ('DRAFT','PREPARED','REVIEWED','APPROVED','LOCKED')),
  content jsonb not null default '{}'::jsonb,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create index ix_workpapers_engagement_phase on audit_workpapers(engagement_id, phase_id);

create table audit_evidence (
  id uuid primary key default uuid_generate_v4(),
  engagement_id uuid not null references audit_engagements(id) on delete cascade,
  workpaper_id uuid not null references audit_workpapers(id) on delete cascade,
  file_name text not null,
  mime_type text,
  file_size bigint,
  storage_path text not null,
  hash_sha256 text,
  uploaded_by uuid references platform_users(id),
  uploaded_at timestamptz not null default now()
);

create table audit_signoff_logs (
  id uuid primary key default uuid_generate_v4(),
  firm_id uuid not null references platform_firms(id),
  engagement_id uuid references audit_engagements(id) on delete cascade,
  entity_type text not null check (entity_type in ('PHASE','WORKPAPER','FWC')),
  entity_id uuid not null,
  prepared_by uuid references platform_users(id),
  prepared_at timestamptz,
  reviewed_by uuid references platform_users(id),
  reviewed_at timestamptz,
  approved_by uuid references platform_users(id),
  approved_at timestamptz,
  status text not null default 'DRAFT' check (status in ('DRAFT','PREPARED','REVIEWED','APPROVED','LOCKED')),
  ip_address inet,
  user_agent text
);

create index ix_signoff_entity on audit_signoff_logs(entity_type, entity_id);

-- ========= FIRM WIDE CONTROLS (ISQM) =========

create table fwc_domains (
  id uuid primary key default uuid_generate_v4(),
  firm_id uuid not null references platform_firms(id),
  domain_code text not null,
  domain_name text not null,
  unique(firm_id, domain_code)
);

create table fwc_records (
  id uuid primary key default uuid_generate_v4(),
  firm_id uuid not null references platform_firms(id),
  domain_id uuid not null references fwc_domains(id) on delete cascade,
  record_type text not null check (record_type in ('POLICY','CHECKLIST','REGISTER','PLAN','REPORT','DEFICIENCY')),
  title text not null,
  status text not null default 'DRAFT' check (status in ('DRAFT','PREPARED','REVIEWED','APPROVED','LOCKED')),
  version_no int not null default 1,
  data jsonb not null default '{}'::jsonb,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create table fwc_version_logs (
  id uuid primary key default uuid_generate_v4(),
  record_id uuid not null references fwc_records(id) on delete cascade,
  version_no int not null,
  change_summary text,
  changed_by uuid references platform_users(id),
  changed_at timestamptz not null default now()
);

create table monitoring_deficiencies (

  record_id uuid references fwc_records(id) on delete set null,
  severity text not null check (severity in ('LOW','MEDIUM','HIGH')),
  root_cause text,
  remediation_plan text,
  target_date date,
  status text not null default 'OPEN' check (status in ('OPEN','IN_PROGRESS','REMEDIATED','VERIFIED','CLOSED')),
  owner_user_id uuid references platform_users(id),
  created_at timestamptz not null default now()
);

create table eqcr_assignments (
  id uuid primary key default uuid_generate_v4(),
  engagement_id uuid not null references audit_engagements(id) on delete cascade,
  eqcr_user_id uuid not null references platform_users(id),
  status text not null default 'ASSIGNED' check (status in ('ASSIGNED','IN_REVIEW','COMPLETED')),
  completed_at timestamptz
);

create table audit_export_jobs (
  id uuid primary key default uuid_generate_v4(),
  firm_id uuid not null references platform_firms(id),
  engagement_id uuid references audit_engagements(id) on delete cascade,
  export_type text not null check (export_type in ('INSPECTION_PACK','FS','WORKPAPERS')),
```
